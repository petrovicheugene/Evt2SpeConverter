266 void ZBaseMainWindow::zh_appLanguageControl()
267 {
268  if (!sender())
269  {
270  return;
271  }
272 
273  QAction* action = dynamic_cast<QAction*>(sender());
274  if (!action)
275  {
276  return;

278 
279  ZTranslatorManager translatorManager;
280  QString currentLanguageName;
281  bool ok = false;
282  translatorManager.zp_currentLanguageName(currentLanguageName, &ok);
283 
284  if (ok && action->text() == currentLanguageName)
285  {
286  return;
287  }
288 
289  ok = false;
290  translatorManager.zp_setApplicationLanguage(action->text(), &ok);
291  QString msg;
292  if (ok)

295  "In order for the changes to take effect, please restart the application.");
296  } else
297  {
298  msg = translatorManager.zp_lastError().isEmpty() ? tr("Unknown language settings error.")
299  : translatorManager.zp_lastError();
300  }
301 
302  QString title = tr("Language settings");
303  QMessageBox::information(this, title, msg, QMessageBox::Ok);
304 }
